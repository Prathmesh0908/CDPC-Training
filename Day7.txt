*Product of Array Self*
def productExceptSelf(nums):
    result = []
    for i in range(len(nums)):
        prod = 1
        for j in range(len(nums)):
            if i != j:
                prod *= nums[j]
        result.append(prod)   
    return result
arr = [1, 2, 3, 4]
print(productExceptSelf(arr))
